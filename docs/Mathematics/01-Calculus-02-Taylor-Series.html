from ipywidgets import *
import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline

x = np.arange(0, 100, 1)

def update(w = 1.0):
    fig = plt.figure()
    ax = fig.add_subplot(1, 1, 1)
    ax.plot(x, x**w)
    ax.plot(x, x**1)
    fig.canvas.draw()

interact(update);
